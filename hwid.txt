    <button onclick="validateHWID()">HWID'i Doğrula</button><br><br>
    <p id="validationResult"></p>

    <script>
        async function validateHWID() {
            let encryptedHwid = document.getElementById("encryptedHwid").value.trim();
            let url = "https://yourusername.github.io/hwid"; // GitHub Pages URL'nizi buraya ekleyin

            try {
                let response = await fetch(url, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ hwid: encryptedHwid })
                });

                if (response.ok) {
                    let responseData = await response.json();
                    if (responseData.status === "valid") {
                        document.getElementById("validationResult").innerText = "HWID doğrulandı!";
                    } else {
                        document.getElementById("validationResult").innerText = "HWID geçersiz!";
                    }
                } else {
                    throw new Error('Network response was not ok.');
                }
            } catch (error) {
                console.error('There has been a problem with your fetch operation:', error);
                document.getElementById("validationResult").innerText = "Bir hata oluştu: " + error.message;
            }
        }
    </script>
